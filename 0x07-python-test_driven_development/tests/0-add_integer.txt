#!/usr/bin/python3
def test_add_integer():
    """
    This function tests the 'add_integer' function.
    """
    assert add_integer(1, 1) == 2
    assert add_integer(1.5, 2.5) == 3
    assert add_integer(100) == 198
    try:
        add_integer("1", 2)
    except TypeError as e:
        assert str(e) == "a must be an integer"
    try:
        add_integer(1, "2")
    except TypeError as e:
        assert str(e) == "b must be an integer"
